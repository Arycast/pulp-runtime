## Pulp Runtime

pulp_runtime.path={build.core.path}/pulp-runtime
pulp_runtime.include.path={pulp_runtime.path}/include

pulp_runtime_libc.path={pulp_runtime.path}/lib/libc/minimal
pulp_runtime_libc.include.path={pulp_runtime_libc.path}/include

pulp_runtime.soc=pulpissimo


## Toolchain global definitions

compiler.path={runtime.tools.pulpissimo_toolchain_openocd.path}/bin

compiler.warning_flags=-Wunused -Wuninitialized
compiler.warning_flags.none=
compiler.warning_flags.default=-Wall
compiler.warning_flags.more=-Wall -Wextra
compiler.warning_flags.all=-Wall -Wextra -Wconversion -pedantic


## Preprocessing

compiler.arch_flags=-march=rv32imfcxpulpv2
arduino_related_defines=-DARDUINO={runtime.ide.version} -DARDUINO_{build.board} -DARDUINO_ARCH_{build.arch}
compiler.define_flags=-D__riscv__ -D__PLATFORM__=ARCHI_PLATFORM_FPGA -DARCHI_FPGA_PER_FREQUENCY=12400000 -DARCHI_FPGA_SOC_FREQUENCY=10000000 -DCONFIG_IO_UART=1 -DCONFIG_IO_UART_BAUDRATE=115200 -DCONFIG_IO_UART_ITF=0 {arduino_related_defines}
compiler.include_flags=-I{pulp_runtime.include.path} -I{pulp_runtime.include.path}/chips/{pulp_runtime.soc} -I{pulp_runtime.include.path}/archi/chips/{pulp_runtime.soc} -I{pulp_runtime.include.path}/hal/chips/{pulp_runtime.soc} -I{pulp_runtime_libc.include.path} -include {pulp_runtime.include.path}/chips/pulpissimo/config.h


## Compiler

compiler.S.cmd=riscv32-unknown-elf-gcc
compiler.S.cmd.option_flags=-fdata-sections -ffunction-sections -fno-jump-tables -fno-tree-loop-distribute-patterns
compiler.S.flags={compiler.arch_flags} {compiler.S.cmd.option_flags} {compiler.define_flags} -DLANGUAGE_ASSEMBLY {compiler.include_flags} -c -ggdb

compiler.c.cmd=riscv32-unknown-elf-gcc
compiler.c.cmd.option_flags=-fdata-sections -ffunction-sections -fno-jump-tables -fno-tree-loop-distribute-patterns
compiler.c.flags={compiler.warning_flags} {compiler.arch_flags} {compiler.c.cmd.option_flags} {compiler.define_flags} {compiler.include_flags} -c -ggdb

compiler.cpp.cmd=riscv32-unknown-elf-g++
compiler.cpp.cmd.option_flags=-fdata-sections -ffunction-sections -fno-jump-tables -fno-tree-loop-distribute-patterns -fnothrow-opt -fno-rtti -fno-exceptions
compiler.cpp.flags={compiler.warning_flags} {compiler.arch_flags} {compiler.cpp.cmd.option_flags} {compiler.define_flags} {compiler.include_flags} -c -ggdb

compiler.ar.cmd=riscv32-unknown-elf-gcc-ar
compiler.ar.flags=rcsPv


## Linking

compiler.c.elf.cmd.option_flags=
compiler.c.elf.cmd=riscv32-unknown-elf-g++
compiler.c.elf.flags={compiler.arch_flags} {compiler.c.elf.cmd.option_flags} -nostartfiles -nostdlib -Wl,--gc-sections,--print-gc-sections,--print-memory-usage,--verbose=4 -ggdb -T{pulp_runtime.path}/kernel/chips/pulpissimo/link.ld

compiler.libraries.ldflags=-L{build.path}
compiler.libraries.libs=-lstdc++ -lm -lgcc -lc


## ELF processing

compiler.objcopy.cmd=riscv32-unknown-elf-objcopy

compiler.objdump.cmd=riscv32-unknown-elf-objdump

compiler.elf2hex.cmd=riscv32-unknown-elf-objcopy

compiler.size.cmd=riscv32-unknown-elf-size


## Compile S files
recipe.S.o.pattern="{compiler.path}/{compiler.S.cmd}" {compiler.S.flags} {includes} "{source_file}" -o "{object_file}"

## Compile c files
recipe.c.o.pattern="{compiler.path}/{compiler.c.cmd}" {compiler.c.flags} {includes} "{source_file}" -o "{object_file}"

## Compile c++ files
recipe.cpp.o.pattern="{compiler.path}/{compiler.cpp.cmd}" {compiler.cpp.flags} {includes} "{source_file}" -o "{object_file}"

## Create archives
recipe.ar.pattern="{compiler.path}/{compiler.ar.cmd}" {compiler.ar.flags} "{archive_file_path}" "{object_file}"


## Combine gc-sections, archives, and objects
recipe.c.combine.pattern="{compiler.path}/{compiler.c.elf.cmd}" {compiler.c.elf.flags} {compiler.libraries.ldflags} -o "{build.path}/{build.project_name}.elf" {object_files} "{archive_file_path}" {compiler.libraries.libs}
