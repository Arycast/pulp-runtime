## Pulp Runtime

pulp_runtime.path={runtime.platform.path}/libraries/pulp-runtime/src
pulp_runtime.soc=pulpissimo

## Toolchain global definitions

compiler.path={runtime.tools.pulpissimo_toolchain_openocd.path}/bin

compiler.warning_flags=-Wunused -Wuninitialized
compiler.warning_flags.none=
compiler.warning_flags.default=-Wall
compiler.warning_flags.more=-Wall -Wextra
compiler.warning_flags.all=-Wall -Wextra -Wconversion -pedantic


## Preprocessing

compiler.define_flags=-D__riscv__ -D__PLATFORM__=ARCHI_PLATFORM_FPGA -DARCHI_FPGA_PER_FREQUENCY=12400000 -DARCHI_FPGA_SOC_FREQUENCY=10000000 -DCONFIG_IO_UART=1 -DCONFIG_IO_UART_BAUDRATE=115200 -DCONFIG_IO_UART_ITF=0
compiler.arch_flags=-march=rv32imfcxpulpv2
compiler.include_flags=-I{pulp_runtime.path}/include -I{pulp_runtime.path}/include/chips/{pulp_runtime.soc} -I{pulp_runtime.path}/include/archi -I{pulp_runtime.path}/include/archi/chips/{pulp_runtime.soc} -I{pulp_runtime.path}/include/hal -I{pulp_runtime.path}/include/hal/chips/{pulp_runtime.soc} -include {pulp_runtime.path}/include/chips/pulpissimo/config.h

## Compiler

compiler.S.cmd=riscv32-unknown-elf-gcc
compiler.S.cmd.option_flags=-fdata-sections -ffunction-sections -fno-jump-tables -fno-tree-loop-distribute-patterns -flto
compiler.S.flags={compiler.arch_flags} {compiler.S.cmd.option_flags} {compiler.define_flags} -DLANGUAGE_ASSEMBLY {compiler.S.cmd.option_flags} {compiler.include_flags} -c -ggdb

compiler.c.cmd=riscv32-unknown-elf-gcc
compiler.c.cmd.option_flags=-fdata-sections -ffunction-sections -fno-jump-tables -fno-tree-loop-distribute-patterns -flto
compiler.c.flags={compiler.warning_flags} {compiler.arch_flags} {compiler.c.cmd.option_flags} {compiler.define_flags} {compiler.c.cmd.option_flags} {compiler.include_flags} -c -ggdb

compiler.cpp.cmd=riscv32-unknown-elf-g++
compiler.cpp.cmd.option_flags=-fdata-sections -ffunction-sections -fno-jump-tables -fno-tree-loop-distribute-patterns -fnothrow-opt -fno-rtti -fno-exceptions -flto
compiler.cpp.flags={compiler.warning_flags} {compiler.arch_flags} {compiler.cpp.cmd.option_flags} {compiler.define_flags} {compiler.cpp.cmd.option_flags} {compiler.include_flags} -c -ggdb

compiler.ar.cmd=riscv32-unknown-elf-ar
compiler.ar.flags=rcs


## Linking

compiler.c.elf.cmd.option_flags=-flto -fuse-linker-plugin
compiler.c.elf.cmd=riscv32-unknown-elf-g++
compiler.c.elf.flags=-lto -fuse-linker-plugin

compiler.libraries.ldflags=

## ELF processing

compiler.objcopy.cmd=riscv32-unknown-elf-objcopy

compiler.objdump.cmd=riscv32-unknown-elf-objdump

compiler.elf2hex.cmd=riscv32-unknown-elf-objcopy

compiler.size.cmd=riscv32-unknown-elf-size


## Compile S files
recipe.S.o.pattern="{compiler.path}/{compiler.S.cmd}" {compiler.S.flags} {includes} "{source_file}" -o "{object_file}"

## Compile c files
recipe.c.o.pattern="{compiler.path}/{compiler.c.cmd}" {compiler.c.flags} {includes} "{source_file}" -o "{object_file}"

## Compile c++ files
recipe.cpp.o.pattern="{compiler.path}/{compiler.cpp.cmd}" {compiler.cpp.flags} {includes} "{source_file}" -o "{object_file}"

## Create archives
recipe.ar.pattern="{compiler.path}/{compiler.ar.cmd}" {compiler.ar.flags} "{archive_file_path}" "{object_file}"


## Combine gc-sections, archives, and objects
recipe.c.combine.pattern="{compiler.path}/{compiler.c.elf.cmd}" {compiler.c.elf.flags} -o "{build.path}/{build.project_name}.elf" {object_files} {compiler.libraries.ldflags} "{archive_file_path}" "-L{build.path}"


# support for debugging

# If the debug.executable property is not present or is empty debugging will not be allowed.
# absolute path to the compiled binary of the sketch
debug.executable={build.path}/{build.project_name}.elf

# a unique identifier of the required toolchain, currently we support gcc (and compatible) only
debug.toolchain=gdb

# absolute path to the toolchain directory
debug.toolchain.path={runtime.tools.pulpissimo_toolchain_openocd.path}/bin

# prefix of the toolchain (for example arm-none-eabi-)
debug.toolchain.prefix=riscv32-unknown-elf-


# a unique identifier of the required debug server, currently we support only openocd
debug.server=openocd
# absolute path to the SVD descriptor.
#debug.svd_file=

# absolute path to the OpenOCD directory
debug.server.openocd.path={runtime.tools.pulpissimo_toolchain_openocd.path}/bin
